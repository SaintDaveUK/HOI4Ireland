#ULS_set_derry_name_effect = yes
ULS_set_derry_name_effect = {
	
	if = {
		limit = {
			7442 = { is_controlled_by = ROOT }
		}
		#DERRY
		ULS_to_derry_effect = yes #default
		
		#LONDONDERRY override
		if = {
			limit = {
				OR = {
					original_tag = ENG
					AND = {
						original_tag = ULS
						AND = {
							NOT = { is_subject_of = IRE }
							NOT = { has_government = neutrality }
						}
					}
				}
			}
			ULS_to_londonderry_effect = yes
		}
		
		#FREE DERRY override
		if = {
			limit = {
				has_government = neutrality
				OR = {
					original_tag = IRE
					original_tag = ULS
				}
			}
			ULS_to_free_derry_effect = yes
		}
	}
}
